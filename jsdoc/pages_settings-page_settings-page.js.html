import '/node_modules/@polymer/polymer/polymer-legacy.js';

import '/node_modules/@polymer/iron-flex-layout/iron-flex-layout-classes.js';
import '/node_modules/@polymer/paper-styles/typography.js';
import '/node_modules/@polymer/paper-styles/color.js';
import '/node_modules/@polymer/app-layout/app-toolbar/app-toolbar.js';
import '/node_modules/@polymer/paper-material/paper-material.js';
import '/elements/setting-elements/setting-toggle/setting-toggle.js';
import '/elements/setting-elements/setting-dropdown/setting-dropdown.js';
import '/elements/setting-elements/setting-text/setting-text.js';
import '/elements/error-dialog/error-dialog.js';
import { Polymer } from '/node_modules/@polymer/polymer/lib/legacy/polymer-fn.js';
import { html } from '/node_modules/@polymer/polymer/lib/utils/html-tag.js';

import '/scripts/chrome-extension-utils/scripts/ex_handler.js';

(function(window) {

  const _ERR_SETTING_CHANGE = 'Failed to change setting';

  window.app = window.app || {};

  /**
   * Polymer element to manage extension settings
   * @namespace SettingsPage
   */
  app.SettingsPageFactory = Polymer({
    _template: html`
  <style include="iron-flex iron-flex-alignment shared-styles"></style>
    <style>
      :host {
        display: block;
        position: relative;
      }

      .body-content {
        height: 100%;
        padding-left: 16px;
        padding-right: 16px;
      }
    </style>

    <paper-material elevation="1" class="page-content vertical layout">

      <!-- Tool bar -->
      <paper-material elevation="1">
        <app-toolbar class="page-toolbar">
          <span class="space" slot="top"></span>
          <div class="middle middle-container center horizontal layout flex">
            <div class="flex">Settings</div>
          </div>
        </app-toolbar>
      </paper-material>

      <!-- Content -->
      <div class="body-content">
        
        <!-- Error dialog -->
        <error-dialog id="errorDialog"></error-dialog>

        <setting-toggle section-title="Clipboard &amp; messaging" name="monitorClipboard" main-label="Monitor clipboard" secondary-label="Automatically store contents when possible">
        </setting-toggle>
        <setting-toggle name="allowPush" main-label="Push to devices" secondary-label="Send messages to your registered devices" disabled\$="[[!isSignedIn]]" checked="{{pushEnabled}}">
        </setting-toggle>
        <setting-toggle name="autoSend" main-label="Auto send" secondary-label="Automatically send when possible" disabled\$="[[_computeAutoSendDisabled(isSignedIn, pushEnabled)]]" indent="">
        </setting-toggle>
        <setting-toggle name="allowReceive" main-label="Receive from devices" disabled\$="[[!isSignedIn]]" secondary-label="Receive messages from your registered devices">
        </setting-toggle>
        <setting-toggle name="highPriority" main-label="High priority message" secondary-label="Wakes android devices for immediate delivery. Uses more battery." disabled\$="[[!isSignedIn]]" noseparator="">
        </setting-toggle>

        <setting-text section-title="Device" id="deviceNickname" name="deviceNickname" main-label="Nickname" secondary-label="Descriptive name for this device" placeholder="e.g. Home Laptop" maxlength="32">
        </setting-text>
        <setting-dropdown name="storageDuration" label="Storage duration" items="[&quot;One day&quot;, &quot;One week&quot;, &quot;One month&quot;,
             &quot;One year&quot;, &quot;Until space needed&quot;]" noseparator="">
        </setting-dropdown>
        <setting-toggle id="autoBackup" name="autoBackup" main-label="Auto data backup" secondary-label="Perform a daily backup of all your data to Google Drive" on-change="_onAutoBackupChanged" disabled\$="[[!isSignedIn]]" noseparator="">
        </setting-toggle>

        <setting-toggle section-title="Notifications" name="notifyOnCopy" main-label="Notify on clipboard change" secondary-label="Display notification when a copy/cut is detected">
        </setting-toggle>
        <setting-toggle name="notifyOnSend" main-label="Notify on send" secondary-label="Display notification when a message is sent" disabled\$="[[!isSignedIn]]">
        </setting-toggle>
        <setting-toggle name="notifyOnError" main-label="Notify on error (recommended)" secondary-label="Display notification when an error occurs" noseparator="">
        </setting-toggle>
      </div>

    </paper-material>
`,

    is: 'settings-page',

    properties: {
      /**
       * Is user signed in
       * @memberOf SettingsPage
       */
      isSignedIn: {
        type: Boolean,
        value: false,
        notify: true,
      },
    },

    /**
     * Element is ready
     * @memberOf SettingsPage
     */
    ready: function() {
      // listen for chrome messages
      Chrome.Msg.listen(this._onChromeMessage.bind(this));
    },

    /**
     * We have animated in and are now the current page
     * @memberOf SettingsPage
     */
    onCurrentPage: function() {
      this.set('isSignedIn', app.Utils.isSignedIn());
    },

    /**
     * Event: Fired when a message is sent from either an extension process<br>
     * (by runtime.sendMessage) or a content script (by tabs.sendMessage).
     * @see https://developer.chrome.com/extensions/runtime#event-onMessage
     * @param {Chrome.Msg.Message} request - details for the
     * @param {string} request.error - error message
     * @returns {boolean} true if asynchronous
     * @private
     * @memberOf SettingsPage
     */
    _onChromeMessage: function(request) {
      let ret = false;
      if (request.message === app.ChromeMsg.REGISTER_FAILED.message) {
        this._showDialog(_ERR_SETTING_CHANGE, request.error);
      } else if (request.message === app.ChromeMsg.UNREGISTER_FAILED.message) {
        this._showDialog(_ERR_SETTING_CHANGE, request.error);
      }
      return ret;
    },

    /**
     * Event: Auto backup changed by user
     * @param {Event} event
     * @private
     * @memberOf SettingsPage
     */
    _onAutoBackupChanged: function(event) {
      if (event.target.checked) {
        app.Drive.addScope().catch((err) => {
          this.$.autoBackup.setChecked(false);
          this._showDialog(_ERR_SETTING_CHANGE, err.message);
        });
      }
    },

    /**
     * Computed Binding: Is autoSend setting disabled
     * @param {boolean} isSignedIn
     * @param {boolean} allowPush
     * @returns {boolean} true if disabled
     * @private
     * @memberOf SettingsPage
     */
    _computeAutoSendDisabled: function(isSignedIn, allowPush) {
      return !(isSignedIn && allowPush);
    },

    /**
     * Display error dialog
     * @param {string} title - error title
     * @param {string} text - description of error
     * @private
     * @memberOf SettingsPage
     */
    _showDialog: function(title, text) {
      this.$.errorDialog.open(title, text);
    },
  });
})(window);
